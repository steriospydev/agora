- PayeeType
    id: uuid4 PK
    label: char(50) Unique

- Payee
    id : uuid4 PK
    label_id: FK -> PayeeType
    name: char(100)
    summary: char(250) Null
    active: bool, default=False

- Payment
    id: uuid4 PK
    payee: FK -> Payee
    amount: float, default=0.00
    summary: char(250) Null
    payment_day: date
    paid: bool
